79       /**
80        * Método para manejar el evento de añadir una tarea.
81        * Se encarga de obtener los datos de la tarea desde la interfaz de usuario,
82        * validarlos, agregar la tarea a la base de datos y asignarla al usuario seleccionado.
83        *
84        * @param actionEvent Evento que desencadena la acción de añadir tarea.
85        */
86   
87       public void anyadirTarea(ActionEvent actionEvent) {
88           // Obtener datos de la tarea desde la interfaz de usuario
89           String nota;
90           String titulo;
91           String recomp;
92   
93           nota = notaUser.getText();
94           titulo = TituloTi.getText();
95           recomp = recompensa.getText();
96   
97           // Verificar si el título está vacío
98           if (titulo == null || titulo.trim().isEmpty()) {
99               // Mostrar alerta si el título está vacío
100              Alert alert = new Alert(Alert.AlertType.INFORMATION);
101              alert.setContentText("Nombre de Tarea Vacía");
102              alert.showAndWait();
103              return;
104          }
105  
106          // Obtener el usuario seleccionado del ChoiceBox
107          Usuarios usuarioSeleccionado = (Usuarios) usuariosChoiceBox.getValue();
108          if (usuarioSeleccionado == null) {
109              // Mostrar alerta si no se ha seleccionado un usuario
110              showAlert(Alert.AlertType.ERROR, "Usuario no seleccionado", "Debe seleccionar un usuario para asignar la tarea.");
111              return;
112          }
113  
114          Alert alert = new Alert(Alert.AlertType.INFORMATION);
115  
116          // Verificar si se ha seleccionado añadir recompensa
117          if (anyadirRecompensaCh.isSelected()) {
118              // Verificar si la recompensa ingresada es un número válido
119              if (isValidNumber(recomp)) {
120                  // Agregar la tarea con recompensa
121                  TareaModel TM = new TareaModel();
122                  TM.anyadirTarea1(titulo, nota, recomp);
123              } else {
124                  // Mostrar alerta si la recompensa no es un número válido
125                  showAlert(Alert.AlertType.ERROR, "Valor Inválido", "Por favor, ingrese un número válido para la recompensa.");
126                  return;
127              }
128          } else {
129              // Agregar la tarea sin recompensa
130              TareaModel TM = new TareaModel();
131              TM.anyadirTarea2(titulo, nota);
132          }
133  
134          // Recuperar datos de usuario y grupo
135          recuperarDatosUsuario();
136          recuperarDatosGrupo();
137          TareaModel T = new TareaModel();
138          // Obtener el ID de la tarea recién agregada
139          int idTarea = T.getVerIdTarea(titulo);
140          try {
141              // Asignar la tarea al usuario seleccionado en el grupo actual
142              T.asignarTarea(g.getIdGrupo(), usuarioSeleccionado.getId(), idTarea,false);
143          } catch (SQLException e) {
144              // Manejar excepción de SQL
145              throw new RuntimeException(e);
146          }
147  
148          // Mostrar mensaje de éxito
149          alert.setTitle("Info");
150          alert.setHeaderText("Estado");
151          alert.setContentText("Tarea guardada con éxito");
152          alert.showAndWait();
153  
154          // Transición para cargar otra vista después de un breve retraso
155          PauseTransition pause = new PauseTransition(Duration.seconds(2));
156          pause.setOnFinished(e -> {
157              try {
158                  // Cargar otra vista después del retraso
159                  AnchorPane pane = FXMLLoader.load(getClass().getResource("Vistas-admin/TareasRealizarGrupo-view.fxml"));
160                  this.PanelAnyadirTareas.getChildren().setAll(pane);
161              } catch (IOException ex) {
162                  // Manejar excepción de IO
163                  Logger.getLogger(RecuperarContraseñaController.class.getName()).log(Level.SEVERE, null, ex);
164              }
165          });
166          pause.play();
167      }

169      /**
170       * Método para obtener una lista de usuarios que pertenecen al grupo actual.
171       * Realiza una consulta a la base de datos para obtener la información.
172       *
173       * @return Lista de usuarios del grupo actual.
174       */
175      public List<Usuarios> obtenerUsuariosDelGrupo() {
176          recuperarDatosGrupo();
177          List<Usuarios> usuarios = new ArrayList<>();
178          // Consulta SQL para obtener usuarios del grupo actual
179          String sql = "SELECT u.id_usuario, u.nombre FROM usuarios u INNER JOIN grupousuario gu ON u.id_usuario = gu.id_usuario WHERE gu.idgrupo = ?";
180          try (Connection connection = getConexion();
181               PreparedStatement statement = connection.prepareStatement(sql)) {
182              statement.setInt(1, g.getIdGrupo());
183              try (ResultSet resultSet = statement.executeQuery()) {
184                  while (resultSet.next()) {
185                      // Obtener datos de usuario de la consulta
186                      int idUsuario = resultSet.getInt("id_usuario");
187                      String nombre = resultSet.getString("nombre");
188                      // Crear objeto Usuario y añadirlo a la lista
189                      Usuarios usuario = new Usuarios(idUsuario, nombre);
190                      usuarios.add(usuario);
191                  }
192              }
193          } catch (SQLException e) {
194              // Manejar excepción de SQL
195              e.printStackTrace();
196              // Manejo de excepciones según tu caso
197          }
198          // Retornar la lista de usuarios
199          return usuarios;
200      }

214          TextFormatter<String> textFormatter = new TextFormatter<>(change -> {

216                  return change;
217              } else {
218                  return null;
219              }
220          });
221  
222          recompensa.setTextFormatter(textFormatter);
223          cargarUsuariosEnChoiceBox();
224      }
