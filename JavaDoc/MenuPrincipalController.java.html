80       private void cargarImagenUsuario() {
81           // Conexión a la base de datos
82           try (Connection connection = getConexion()) {
83               String sql = "SELECT imagen_Perfil FROM usuarios WHERE id_usuario = ?";
84               try (PreparedStatement statement = connection.prepareStatement(sql)) {
85                   int userId = u.getId();
86                   statement.setInt(1, userId);
87                   ResultSet resultSet = statement.executeQuery();
88   
89                   // Procesar el resultado
90                   if (resultSet.next()) {
91                       byte[] imageData = resultSet.getBytes("imagen_Perfil");
92                       if (imageData != null && imageData.length > 0) {
93                           try (InputStream inputStream = new ByteArrayInputStream(imageData)) {
94                               Image image = new Image(inputStream);
95                               // Verificar si la imagen se ha cargado correctamente
96                               if (image.isError()) {
97                                   System.err.println("Error al cargar la imagen: " + image.getException());
98                               } else {
99                                   // Ajustar la imagen dentro del círculo
100                                  circulo.setFill(new ImagePattern(image));
101                                  System.out.println("Imagen de perfil cargada correctamente.");
102                              }
103                          } catch (IOException e) {
104                              e.printStackTrace();
105                              System.err.println("Error al convertir los datos de la imagen.");
106                          }
107                      } else {
108                          System.out.println("La columna imagen_Perfil está vacía para el usuario con ID: " + userId);
109                      }
110                  } else {
111                      System.out.println("No se encontró ningún usuario con ID: " + userId);
112                  }
113              }
114          } catch (SQLException e) {
115              e.printStackTrace();
116              System.err.println("Error al acceder a la base de datos.");
117          }
