18        * Encripta la contraseña utilizando el algoritmo MD5.
19        *
20        * @param contraseña Contraseña a encriptar.
21        * @return Contraseña encriptada.
22        * @throws NoSuchAlgorithmException Si no se encuentra el algoritmo de encriptación.
23        */
24       private String encriptarContraseña(String contraseña) throws NoSuchAlgorithmException {
25           MessageDigest digest = MessageDigest.getInstance("MD5");
26           byte[] hash = digest.digest(contraseña.getBytes());
27           StringBuilder hexString = new StringBuilder();
28           for (byte b : hash) {
29               String hex = Integer.toHexString(0xff & b);
30               if (hex.length() == 1) hexString.append('0');
31               hexString.append(hex);
32           }
33           return hexString.toString();
34       }

49               String sql = "SELECT id_usuario, nombre FROM usuarios WHERE correo=? AND password=?";
50               PreparedStatement ps = this.getConexion().prepareStatement(sql);
51               ps.setString(1, email);
52               ps.setString(2, contraseñaEncriptada); // Encriptamos la contraseña antes de realizar la consulta
53               ResultSet rs = ps.executeQuery();
54   
55               while (rs.next()) {
56                   int idUsuario = rs.getInt("id_usuario");
57                   String nombreUsuario = rs.getString("nombre");
58   
59                   u = new Usuarios(idUsuario, nombreUsuario, email, contraseñaEncriptada);
60               }
61              // u = new Usuarios(12, "q", "q", "q");
62           } catch (SQLException | NoSuchAlgorithmException e) {
63               e.printStackTrace();

66           }

109       * Verifica si una contraseña ya ha sido utilizada anteriormente.
110       *
111       * @param nuevaContraseña Nueva contraseña a verificar.
112       * @return true si la contraseña ya ha sido utilizada anteriormente, de lo contrario false.
113       */
114      public boolean verificarContraseñaRepetida(String nuevaContraseña) {
115          boolean contraseñaRepetida = false;
116          try {
117              String sql = "SELECT COUNT(*) AS count FROM contraseñasrepetidas WHERE contraseña_anterior = ?";
118              PreparedStatement ps = this.getConexion().prepareStatement(sql);
119              ps.setString(1, nuevaContraseña);
120              ResultSet rs = ps.executeQuery();
121              while (rs.next()) {
122                  int count = rs.getInt("count");
123                  if (count > 0) {
124                      contraseñaRepetida = true;
125                  }
126              }
127          } catch (SQLException e) {
128              e.printStackTrace();
129          } finally {
130              this.cerrarConexion();
131          }
132          return contraseñaRepetida;
