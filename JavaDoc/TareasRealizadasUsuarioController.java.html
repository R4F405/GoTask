79                   // Llama al método 'getInsertar' del controlador para inicializarlo con los datos del grupo actual.
80                   prueba.Insertar(t.getIdTarea(),t.getTitulo());
81                   prueba.setParentController(this);
82                   // Controla la disposición de los elementos en el GridPane.
83   
84                   // Incrementa la columna para el próximo elemento.
85                   row ++;
86                   // Añade el AnchorPane al GridPane en la posición especificada por 'column' y 'row'.
87                   TareasRealizadasUsuarioGridPane.add(anchorPane, column, row); //(child,column,row)

91   
92               } catch (IOException e) {
93                   throw new RuntimeException(e);
94               }

96   
97           }

131      /**
132       * Cambia a la vista del menú principal.
133       *
134       * @param event Evento de acción.
135       */
136      public void cambioMenu(ActionEvent event){
137  
138          try {
139              FXMLLoader loader = new FXMLLoader(getClass().getResource("Vistas-paraTodos/MenuPrincipal2-view.fxml"));
140              Parent nuevaEscenaRoot = loader.load();
141  
142              Stage stage = (Stage) ((Node) event.getSource()).getScene().getWindow();
143  
144              // Configurar la nueva escena en el stage
145              Scene nuevaEscena = new Scene(nuevaEscenaRoot);
146              stage.setScene(nuevaEscena);
147              stage.show();
148  
149          } catch (IOException ex) {
150              Logger.getLogger(MenuPrincipalController.class.getName()).log(Level.SEVERE, null, ex);
151          }
152  
153      }
