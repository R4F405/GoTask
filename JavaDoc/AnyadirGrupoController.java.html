62        * Obtiene el nombre del grupo del campo de texto, crea el grupo y añade el usuario como administrador.
63        * Luego, carga la vista del grupo recién creado.
64        *
65        * @param actionEvent el evento que se produce cuando se hace clic en el botón.
66        */
67       @javafx.fxml.FXML
68       public void onCrearGrupoButtonClick(ActionEvent actionEvent) {
69           // Obtener el nombre del grupo del campo de texto
70           String nombre = NombreGrupoTextField.getText();
71   
72           // Verificar si el nombre del grupo está vacío
73           if (nombre == null || nombre.trim().isEmpty()) {
74               Alert alert = new Alert(Alert.AlertType.INFORMATION);
75               alert.setContentText("Nombre de Grupo Vacío");
76               alert.showAndWait();
77               return;
78   
79           }
80   
81           // Crear un nuevo grupo con el nombre proporcionado
82           gm.getCrearGrupo(nombre);
83   
84           // Obtener el ID del grupo recién creado
85           g = gm.getVerIdGrupo(nombre);
86   
87           // Asignar el usuario actual como administrador del nuevo grupo
88           admin = true;
89           gum.GetAnyadirUsuarioNuevoGrupo(g.getIdGrupo(), u.getId(), admin);
90   
91           // Cargar la vista del grupo recién creado
92           try {
93               AnchorPane pane = FXMLLoader.load(getClass().getResource("Vistas-paraTodos/GrupoPagina.fxml"));
94               this.CrearGrupoPAne.getChildren().setAll(pane);
95           } catch (IOException ex) {
96               Logger.getLogger(AnyadirGrupoController.class.getName()).log(Level.SEVERE, null, ex);
97           }
98       }
