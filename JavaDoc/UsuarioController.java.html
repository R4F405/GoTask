61                   AnchorPane anchorPane = fxmlLoader.load();
62                   // Obtiene el controlador asociado con el archivo FXML cargado.
63                   VerUsuariosController prueba = fxmlLoader.getController();
64                   // Llama al método 'getInsertar' del controlador para inicializarlo con los datos del grupo actual.
65                   prueba.insertar(u.getId(),u.getNombre());
66   
67                   // Controla la disposición de los elementos en el GridPane.
68   
69                   // Incrementa la columna para el próximo elemento.
70                   row ++;
71                   // Añade el AnchorPane al GridPane en la posición especificada por 'column' y 'row'.
72                   VerUsuariosGridPane.add(anchorPane, column, row); //(child,column,row)
